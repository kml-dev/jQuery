이벤트 등록 방법 - delegate

이미 만들어져 있는 버튼 class=btn이나 
실행 후에 동적으로 새롭게 만들어진 버튼 class=btn
에 동시에 같은 이벤트 처리기를 등록한다.

$('부모요소 or document').delegate('이벤트 대상 요소','이벤트이름',function(){})
$('부모요소').on('이벤트이름','이벤트 대상 요소',function(){})
$('.dd').on('click','.btn',function(){})
$(document).on('click','.btn', function(){})
